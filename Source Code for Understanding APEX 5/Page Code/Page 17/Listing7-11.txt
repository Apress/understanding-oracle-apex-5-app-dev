-- Listing 7-11. Second Revision of the Get Info Code of Listing 7-5declare    v_valuesToHash apex_t_varchar2;begin    -- First retrieve the data.     select Job, Sal, DeptNo    into :P17_UPDATE_JOB,          :P17_UPDATE_SALARY,          :P17_UPDATE_DEPT    from EMP    where EmpNo = :P17_UPDATE_EMPNO;    -- Then save the hash of these values.    v_valuesToHash := apex_t_varchar2(:P17_UPDATE_JOB, :P17_UPDATE_SALARY,                                      :P17_UPDATE_DEPT);    :P17_HASH := apex_util.get_hash(v_valuesToHash);                                                          end;
